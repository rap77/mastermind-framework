   ```typescript
     autoFollow: boolean; // auto-scroll to newest
     isolatedBrainId: string | null; // if set, show only this brain's logs

     setAutoFollow: (enabled: boolean) => void;
     setIsolatedBrain: (brainId: string | null) => void;
     reset: () => void;
   }
   ```

2. **Implementation:**
   - Use Zustand v5 (no special middleware needed)
   - Persist to localStorage under key `mm_log_filters_{taskId}`
   - Format: JSON string `{ filterLevels: ['info', 'warn', 'error'], autoFollow: true, isolatedBrainId: null }`
   - On init: hydrate from localStorage if exists, otherwise use defaults (all levels on, auto-follow on)

3. **Hydration pattern (handle SSR):**
   ```typescript
   export const useLogFilterStore = create<LogFilterState>((set, get) => {
     // Check localStorage only on client
     const storedState = typeof window !== 'undefined'
       ? JSON.parse(localStorage.getItem('mm_log_filters') || '{}')
       : {};

     return {
       filterLevels: new Set(storedState.filterLevels || ['info', 'warn', 'error']),
       autoFollow: storedState.autoFollow ?? true,
       isolatedBrainId: storedState.isolatedBrainId ?? null,

       toggleLevel: (level) => {
         set((state) => {
           const newSet = new Set(state.filterLevels);
           if (newSet.has(level)) newSet.delete(level);
           else newSet.add(level);

           // Persist
           localStorage.setItem('mm_log_filters', JSON.stringify({
             filterLevels: Array.from(newSet),
             autoFollow: state.autoFollow,
             isolatedBrainId: state.isolatedBrainId,
           }));

           return { filterLevels: newSet };
         });
       },
       // ... other actions
     };
   });
   ```

4. **Defaults:** all levels on, auto-follow enabled, no isolation

5. **Persistence scope:** Per session (not per task), survives page reload but resets when browser closed (no cookies)

   ```typescript
   export function BrainYAMLViewer({ brainId, open, onOpenChange }: BrainYAMLViewerProps) {
     const [yamlContent, setYamlContent] = useState<string>('');
     const [isLoading, setIsLoading] = useState(false);

     useEffect(() => {
       if (!open) return;

       const fetchYAML = async () => {
         setIsLoading(true);
         try {
           const res = await fetch(`/api/brains/${brainId}/yaml`);
           if (!res.ok) throw new Error('Failed to fetch brain YAML');
           const yaml = await res.text();
           setYamlContent(yaml);
         } catch (e) {
           toast.error('Failed to load brain YAML');
         } finally {
           setIsLoading(false);
         }
       };

       fetchYAML();
     }, [open, brainId]);

     const handleCopy = async () => {
       try {
         await navigator.clipboard.writeText(yamlContent);
         toast.success('Copied to clipboard');
       } catch (e) {
         toast.error('Failed to copy to clipboard');
       }
     };

     return (
       <Dialog open={open} onOpenChange={onOpenChange}>
         <DialogContent className="max-w-2xl max-h-[80vh] overflow-auto">
           <DialogHeader>
             <DialogTitle>Brain Configuration: {brainId}</DialogTitle>
           </DialogHeader>

           {isLoading && <p>Loading...</p>}

           {!isLoading && (
             <>
               <SyntaxHighlighter language="yaml" style={atomOneDark} className="rounded">
                 {yamlContent}
               </SyntaxHighlighter>

               <DialogFooter>
                 <Button onClick={handleCopy} variant="outline">
                   Copy to Clipboard
                 </Button>
               </DialogFooter>
             </>
           )}
         </DialogContent>
       </Dialog>
     );
   }
   ```

7. **Error states:**

   - YAML parsing fails → show raw text (fallback)
   - Clipboard denied → show error toast
