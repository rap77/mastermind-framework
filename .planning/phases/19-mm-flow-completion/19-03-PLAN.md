### apps/api/mastermind_cli/mm_flow/checkpoint_writer.py

```python
"""
MM-Flow Checkpoint Writer

Detects write tool calls in recent transcript messages and writes
SESSION-CHECKPOINT.md when changes are detected.

This file is in the repository (NOT ~/.mm-flow/) for:
1. Version control — checkpoint logic evolves with the project
2. CI testability — no Claude Code runtime required
3. Distribution — available on every checkout
"""
import json
import sys
import os
from datetime import datetime
from pathlib import Path

WRITE_TOOL_NAMES = frozenset({"Edit", "Write", "MultiEdit"})
BASH_MUTATION_PATTERNS = [
    " > ", " >> ", "tee ", "touch ", "mkdir ", "mv ", "cp ", "rm ",
]
CHECKPOINT_PATH = Path(".planning/SESSION-CHECKPOINT.md")


def has_write_operations(transcript: list[dict], window: int = 10) -> bool:
    """
    Returns True if any of the last `window` messages contain write tool calls.
    Write = Edit/Write/MultiEdit tool, or Bash with mutation patterns.
    Read-only sessions (grep, cat, ls, read) return False.
    """
    recent = transcript[-window:] if len(transcript) >= window else transcript
    for message in recent:
        tool_calls = message.get("tool_calls") or message.get("tool_use", [])
        if isinstance(tool_calls, list):
            for tc in tool_calls:
                name = tc.get("name", "") if isinstance(tc, dict) else ""
                if name in WRITE_TOOL_NAMES:
                    return True
                if name == "Bash":
                    cmd = ""
                    if isinstance(tc, dict):
                        inp = tc.get("input", {})
                        cmd = inp.get("command", "") if isinstance(inp, dict) else ""
                    if any(p in cmd for p in BASH_MUTATION_PATTERNS):
                        return True
    return False


def write_checkpoint(
    transcript: list[dict],
    checkpoint_path: Path = CHECKPOINT_PATH,
    timestamp: str | None = None,
) -> Path:
    """Write SESSION-CHECKPOINT.md with saved=false."""
    ts = timestamp or datetime.now().isoformat()
    # Extract last few assistant messages as context
    summaries = []
    for msg in reversed(transcript[-20:]):
        if msg.get("role") == "assistant":
            content = msg.get("content", "")
            if isinstance(content, str) and len(content) > 20:
                summaries.append(content[:200])
                if len(summaries) >= 3:
                    break

    checkpoint_path.parent.mkdir(parents=True, exist_ok=True)
    checkpoint_path.write_text(
        f"---\n"
        f"saved: false\n"
        f"last_updated: {ts}\n"
        f"---\n\n"
        f"# Session Checkpoint\n\n"
        f"Auto-generated by mm-flow-stop hook.\n"
        f"Write operations detected in last 10 messages.\n\n"
        f"## Recent Activity\n\n"
        + "\n\n".join(f"> {s}" for s in summaries)
        + "\n\n"
        f"## Action Required\n\n"
        f"Call `mem_session_summary` at start of next session to preserve this work.\n"
    )
    return checkpoint_path


def main():
    """Entry point called by mm-flow-stop.js via stdin JSON."""
    try:
        data = json.load(sys.stdin)
    except (json.JSONDecodeError, EOFError):
        sys.exit(0)  # No data, skip silently

    transcript = data.get("transcript", [])
    if not transcript:
        sys.exit(0)

    if has_write_operations(transcript):
        written = write_checkpoint(transcript)
        print(f"checkpoint: {written}", file=sys.stderr)
    # If no write ops: exit 0 silently (read-only session, no checkpoint needed)


if __name__ == "__main__":
    main()
```

```python
"""Tests for checkpoint_writer.py — behavioral criteria from Brain #7 C6."""
import pytest
from pathlib import Path
from mastermind_cli.mm_flow.checkpoint_writer import has_write_operations, write_checkpoint


# ---- C6: Mock transcripts with Write in position 8 of 10 ----

def make_transcript(write_pos: int | None, total: int = 10) -> list[dict]:
    """Create synthetic transcript. If write_pos given, inject Write tool call there."""
    msgs = []
    for i in range(total):
        if i == write_pos:
            msgs.append({
                "role": "assistant",
                "tool_calls": [{"name": "Write", "input": {"file_path": "foo.py", "content": "x"}}]
            })
        else:
            msgs.append({
                "role": "assistant",
                "tool_calls": [{"name": "Read", "input": {"file_path": "bar.py"}}]
            })
    return msgs


def test_write_in_position_8_triggers_detection():
    """C6: Write tool call in position 8 of 10 must be detected."""
    transcript = make_transcript(write_pos=8, total=10)
    assert has_write_operations(transcript) is True


def test_read_only_transcript_no_trigger():
    """C6: 10/10 Read-only messages must NOT trigger checkpoint."""
    transcript = make_transcript(write_pos=None, total=10)
    assert has_write_operations(transcript) is False


def test_edit_tool_triggers_detection():
    transcript = [{"role": "assistant", "tool_calls": [{"name": "Edit", "input": {}}]}]
    assert has_write_operations(transcript) is True


def test_bash_with_redirect_triggers():
    transcript = [{"role": "assistant", "tool_calls": [
        {"name": "Bash", "input": {"command": "echo foo > output.txt"}}
    ]}]
    assert has_write_operations(transcript) is True


def test_bash_grep_no_trigger():
    transcript = [{"role": "assistant", "tool_calls": [
        {"name": "Bash", "input": {"command": "grep -r pattern src/"}}
    ]}]
    assert has_write_operations(transcript) is False


def test_write_checkpoint_creates_file(tmp_path):
    transcript = make_transcript(write_pos=0, total=3)
    checkpoint = tmp_path / "SESSION-CHECKPOINT.md"
    write_checkpoint(transcript, checkpoint_path=checkpoint)
    assert checkpoint.exists()
    content = checkpoint.read_text()
    assert "saved: false" in content


def test_write_checkpoint_contains_last_updated(tmp_path):
    transcript = make_transcript(write_pos=0, total=1)
    checkpoint = tmp_path / "SESSION-CHECKPOINT.md"
    write_checkpoint(transcript, checkpoint_path=checkpoint, timestamp="2026-04-14T13:00:00")
    assert "2026-04-14T13:00:00" in checkpoint.read_text()
```

```javascript
#!/usr/bin/env node
// MM-Flow Stop Hook — thin dispatcher to checkpoint_writer.py
// Detects write operations in transcript → delegates to Python (testeable in CI)

const { execFileSync } = require('child_process');
const path = require('path');
const fs = require('fs');

let input = '';
const stdinTimeout = setTimeout(() => process.exit(0), 3000);
process.stdin.setEncoding('utf8');
process.stdin.on('data', chunk => input += chunk);
process.stdin.on('end', () => {
  clearTimeout(stdinTimeout);

  // Find project root (where CLAUDE.md lives, searching upward from cwd)
  let projectRoot = process.cwd();
  while (projectRoot !== path.dirname(projectRoot)) {
    if (fs.existsSync(path.join(projectRoot, 'CLAUDE.md'))) break;
    projectRoot = path.dirname(projectRoot);
  }

  const writerPath = path.join(projectRoot, 'apps/api/mastermind_cli/mm_flow/checkpoint_writer.py');
  if (!fs.existsSync(writerPath)) {
    process.exit(0);  // Not a mastermind project or writer not installed
  }

  try {
    // Use execFile (not exec) to avoid shell injection — uv handles venv
    execFileSync('uv', ['run', 'python', writerPath], {
      cwd: path.join(projectRoot, 'apps/api'),
      input: input,
      stdio: ['pipe', 'pipe', 'pipe'],
    });
  } catch (e) {
    // Silent fail — stop hook must not block Claude Code exit
  }
});
```

```javascript
// Stale checkpoint detection
const checkpointPath = path.join(process.cwd(), '.planning/SESSION-CHECKPOINT.md');
if (fs.existsSync(checkpointPath)) {
  const content = fs.readFileSync(checkpointPath, 'utf8');
  const savedMatch = content.match(/^saved:\s*(.+)$/m);
  const tsMatch = content.match(/^last_updated:\s*(.+)$/m);
  const isSaved = savedMatch && savedMatch[1].trim() === 'true';
  if (!isSaved && tsMatch) {
    const lastUpdated = new Date(tsMatch[1].trim());
    const hoursSince = (Date.now() - lastUpdated.getTime()) / (1000 * 60 * 60);
    if (hoursSince > 48) {
      // Blocking warning — stale unsaved checkpoint
      output.previewMessage = `[STALE CHECKPOINT] Sesión anterior sin guardar (${Math.round(hoursSince)}h). Llama mem_session_summary ANTES de continuar.`;
    } else if (hoursSince > 0) {
      output.previewMessage += ' | [CHECKPOINT] Checkpoint sin guardar detectado — considerá llamar mem_session_summary.';
    }
  }
}
```

```javascript
// Write-tool detection (NOT every N messages — only when writes detected)
const recentMsgs = (data.transcript || []).slice(-10);
const hasWriteOps = recentMsgs.some(msg => {
  const tcs = msg.tool_calls || msg.tool_use || [];
  return Array.isArray(tcs) && tcs.some(tc => {
    const name = tc?.name || '';
    if (['Edit', 'Write', 'MultiEdit'].includes(name)) return true;
    if (name === 'Bash') {
      const cmd = tc?.input?.command || '';
      return [' > ', ' >> ', 'tee ', 'touch ', 'mkdir '].some(p => cmd.includes(p));
    }
    return false;
  });
});

if (hasWriteOps) {
  // Inject checkpoint reminder
  const reminder = '[CHECKPOINT] Detecté cambios en archivos recientes. ¿Hiciste decisiones o descubrimientos importantes? Si sí → llamá mem_save AHORA.';
  // Add to output message if not already present
  if (!output.message?.includes('[CHECKPOINT]')) {
    output.message = (output.message || '') + '\n\n' + reminder;
  }
}
```
