## Task 2.1: Create apps/api/mastermind_cli/mm_flow/cli.py

Create a Click CLI app with `execute-phase` command.

```python
import asyncio
import json
import os
import tempfile
import uuid
from datetime import datetime
from pathlib import Path
import click
import asyncpg

RUNTIME_STATE_PATH = Path(".planning/.mm-flow/runtime-state.json")

def _write_runtime_state(execution_id: str, phase: int, moment: str, brain: int, state: str, backend: str) -> None:
    """Atomic write: temp file + rename to avoid partial reads."""
    data = {
        "execution_id": execution_id,
        "phase": phase,
        "current_moment": moment,
        "active_brain": brain,
        "brain_state": state,
        "backend": backend,
        "updated_at": datetime.now().isoformat(),
    }
    RUNTIME_STATE_PATH.parent.mkdir(parents=True, exist_ok=True)
    tmp = Path(tempfile.mktemp(dir=RUNTIME_STATE_PATH.parent))
    tmp.write_text(json.dumps(data, indent=2))
    tmp.rename(RUNTIME_STATE_PATH)

@click.group()
def cli():
    """MM-Flow CLI — lifecycle management for phase execution."""
    pass

@cli.command("execute-phase")
@click.option("--phase", type=int, required=True, help="Phase number")
@click.option("--start", is_flag=True, help="Mark phase as started")
@click.option("--complete", is_flag=True, help="Mark phase as completed")
@click.option("--commit", default=None, help="Git commit hash at completion")
@click.option("--tokens", type=int, default=0, help="Tokens consumed")
@click.option("--summary", default="", help="Execution summary")
def execute_phase(phase, start, complete, commit, tokens, summary):
    """Lifecycle management for phase execution."""
    if start and complete:
        raise click.UsageError("--start and --complete are mutually exclusive")
    if not start and not complete:
        raise click.UsageError("Either --start or --complete is required")

    postgres_url = os.environ.get("DATABASE_URL", "postgresql://postgres:postgres@localhost:5432/mastermind_bd")

    async def _run():
        conn = await asyncpg.connect(postgres_url)
        try:
            if start:
                execution_id = str(uuid.uuid4())
                await conn.execute(
                    """INSERT INTO phase_executions (id, phase_number, status, started_at, triggered_by)
                       VALUES ($1, $2, 'in_progress', NOW(), 'skill')
                       ON CONFLICT DO NOTHING""",
                    execution_id, phase
                )
                backend = os.environ.get("MM_FLOW_BACKEND", "claude")
                _write_runtime_state(execution_id, phase, "EXECUTION_WAVE", 0, "ACTIVE", backend)
                click.echo(f"execution_id:{execution_id}")

            elif complete:
                # Read execution_id from runtime-state.json if not provided
                if RUNTIME_STATE_PATH.exists():
                    state = json.loads(RUNTIME_STATE_PATH.read_text())
                    execution_id = state.get("execution_id", "")
                else:
                    execution_id = ""

                if execution_id:
                    await conn.execute(
                        """UPDATE phase_executions
                           SET status='completed', completed_at=NOW(),
                               git_commit_hash=$2, tokens_consumed=$3, output_summary=$4
                           WHERE id=$1""",
                        execution_id, commit, tokens, summary
                    )
                backend = os.environ.get("MM_FLOW_BACKEND", "claude")
                _write_runtime_state(execution_id or "", phase, "COMPLETED", 0, "IDLE", backend)
                click.echo(f"Phase {phase} marked complete")
        finally:
            await conn.close()

    asyncio.run(_run())

if __name__ == "__main__":
    cli()
```

## Task 2.2: Create apps/api/mastermind_cli/mm_flow/dispatch_engine.py

```python
from __future__ import annotations

import asyncio
import uuid
from typing import Literal

import asyncpg
from pydantic import BaseModel, ConfigDict


class BrainDispatch(BaseModel):
    model_config = ConfigDict(strict=True)
    brain_id: int
    role: str
    model_profile: Literal["quality", "balanced", "budget"]
    is_barrier: bool


class DispatchResult(BaseModel):
    model_config = ConfigDict(strict=True)
    moment: Literal["DISCUSSION", "PLANNING", "EXECUTION_WAVE", "VERIFICATION"]
    parallel: list[BrainDispatch]
    barrier: list[BrainDispatch]
    budget_remaining: int
    execution_id: str


class BudgetExceededError(Exception):
    pass


# Oracle table for SLI-3 unit tests
DISPATCH_ORACLE: dict[str, dict] = {
    "DISCUSSION":     {"parallel_brain_ids": [1, 2, 3], "barrier_brain_ids": [7]},
    "PLANNING":       {"parallel_brain_ids": [4, 5, 6], "barrier_brain_ids": [7]},
    "EXECUTION_WAVE": {"parallel_brain_ids": [7],       "barrier_brain_ids": [], "sequential": True},
    "VERIFICATION":   {"parallel_brain_ids": [7],       "barrier_brain_ids": [], "sequential": True},
}


class DynamicDispatchEngine:
    """Dispatches brains based on phase+moment, reading config.yml and agent_registry."""

    def __init__(self, postgres_url: str):
        self.postgres_url = postgres_url
        from mastermind_cli.mm_flow.config_loader import load_config
        self.config = load_config()

    async def dispatch(
        self,
        phase: int,
        moment: Literal["DISCUSSION", "PLANNING", "EXECUTION_WAVE", "VERIFICATION"],
    ) -> DispatchResult:
        routing = self.config.brain_routing[moment]

        conn = await asyncpg.connect(self.postgres_url)
        try:
            parallel_brains = await self._get_brains_from_registry(conn, routing.brains)
            barrier_brains = await self._get_brains_from_registry(conn, routing.barrier)
        finally:
            await conn.close()

        budget_remaining = self._check_budget(parallel_brains + barrier_brains)

        return DispatchResult(
            moment=moment,
            parallel=[b for b in parallel_brains if not b.is_barrier],
            barrier=barrier_brains,
            budget_remaining=budget_remaining,
            execution_id=str(uuid.uuid4()),
        )

    async def _get_brains_from_registry(
        self, conn: asyncpg.Connection, brain_ids: list[int]
    ) -> list[BrainDispatch]:
        if not brain_ids:
            return []
        rows = await conn.fetch(
            "SELECT brain_id, role, model_quality, is_barrier, token_budget_per_phase, tokens_consumed_total "
            "FROM agent_registry WHERE brain_id = ANY($1::int[])",
            brain_ids,
        )
        return [
            BrainDispatch(
                brain_id=row["brain_id"],
                role=row["role"],
                model_profile=row["model_quality"],
                is_barrier=row["is_barrier"],
            )
            for row in rows
        ]

    def _check_budget(self, brains: list[BrainDispatch]) -> int:
        """Pre-dispatch budget check. Raises BudgetExceededError if >80% consumed."""
        # budget_remaining is a placeholder — real tracking would query agent_registry
        # tokens_consumed_total, but we don't carry that on BrainDispatch for now
        return 100_000  # placeholder: 100K tokens remaining
```
