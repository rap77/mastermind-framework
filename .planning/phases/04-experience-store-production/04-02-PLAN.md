    from pydantic import BaseModel, Field
    from typing import Any, Optional, Dict
    from enum import Enum
    from mastermind_cli.types.interfaces import BrainInput, BrainOutput

    class BrainOutputType(str, Enum):
        """Message types from ROADMAP YAML specification."""

        OUTPUT = "output"  # Normal brain output
        REQUEST = "request"  # Asking another brain for input
        REJECTION = "rejection"  # Brain #7 rejection
        APPROVAL = "approval"  # Brain #7 approval

    class BrainMessage(BaseModel):
        """Internal message content (YAML-based from ROADMAP)."""

        from_brain: str
        to_brain: str
        type: BrainOutputType
        content: dict[str, Any]  # summary, detail, assumptions, dependencies
        task_id: str
        version: str = "2.0.0"

    class BrainEnvelope(BaseModel):
        """Transport envelope with metadata (Hybrid pattern)."""

        message: BrainMessage
        correlation_id: str  # Links related messages in flow
        transport_metadata: Dict[str, Any] = Field(default_factory=dict)  # latency, retries

        @classmethod
        def create(
            cls,
            from_brain: str,
            to_brain: str,
            payload: BrainInput | BrainOutput,
            correlation_id: str,
            task_id: str,
            message_type: BrainOutputType = BrainOutputType.OUTPUT
        ) -> "BrainEnvelope":
            """Factory to create envelope from brain payload."""

            # Convert BrainInput/Output to content dict
            if isinstance(payload, BrainInput):
                content = {
                    "summary": f"Input from {from_brain}",
                    "detail": payload.query,
                    "context": payload.context
                }
            elif isinstance(payload, BrainOutput):
                content = {
                    "summary": payload.content[:200],  # First 200 chars
                    "detail": payload.raw_output,
                    "metadata": payload.metadata
                }
            else:
                content = {"data": payload}

            message = BrainMessage(
                from_brain=from_brain,
                to_brain=to_brain,
                type=message_type,
                content=content,
                task_id=task_id
            )

            return cls(
                message=message,

    ```python
    from mastermind_cli.types.protocol import BrainEnvelope, BrainOutputType
    from mastermind_cli.orchestrator.dependency_resolver import resolve_dependencies
    from typing import Dict

    class StatelessCoordinator:
        """Per-request orchestrator with brain-to-brain message passing."""

        def __init__(self, mcp_client: MCPClient):
            self.mcp_client = mcp_client
            self.message_log: list[BrainEnvelope] = []  # In-memory trace
            self.brain_outputs: Dict[str, BrainOutput] = {}  # brain_id -> output

        async def execute_flow(
            self,
            brief: str,
            brain_ids: list[str],
            mcp_client: MCPClient
        ) -> CoordinatorResponse:
            """Execute brains with message passing (respect DAG dependencies)."""

            # Build FlowConfig from brain_ids (no dependencies = all independent)
            flow_config = FlowConfig(
                flow_id=f"flow-{id(brief)}",
                nodes={brain_id: [] for brain_id in brain_ids}  # TODO: load from flow.yaml
            )

            # Resolve execution order (topological waves)
            execution_waves = resolve_dependencies(flow_config)

            correlation_id = f"corr-{id(brief)}"

            # Execute wave by wave (parallel within wave, sequential between waves)
            for wave in execution_waves:
                # Execute brains in this wave in parallel
                tasks = [
                    self.execute_brain_with_message(
                        brain_id=brain_id,
                        brief=brief,
                        correlation_id=correlation_id,
                        parent_outputs=self._get_parent_outputs(brain_id, flow_config)
                    )
                    for brain_id in wave
                ]
                await asyncio.gather(*tasks)

            return CoordinatorResponse(
                timestamp=datetime.now(timezone.utc),
                status="success",
                plan=[],
                results=[self.brain_outputs[bid] for bid in brain_ids]
            )

        async def execute_brain_with_message(
            self,
            brain_id: str,
            brief: str,
            correlation_id: str,
            parent_outputs: list[BrainOutput]
        ) -> BrainOutput:
            """Execute brain with parent outputs as context."""

            # Build BrainInput with parent outputs
            context = {
                "parent_outputs": [po.model_dump() for po in parent_outputs],
                "correlation_id": correlation_id
            }

            brain_input = BrainInput(
                brain_id=brain_id,
                query=brief,
                context=context
            )

            # Execute brain
            output = await self.execute_brain(brain_id, brain_input, self.mcp_client)

            # Store output for dependent brains
            self.brain_outputs[brain_id] = output

            # Create BrainEnvelope for logging
            envelope = BrainEnvelope.create(
                from_brain=brain_id,
                to_brain="orchestrator",  # Or next brain in DAG
                payload=output,
                correlation_id=correlation_id,
                task_id=f"task-{brain_id}",
                message_type=BrainOutputType.OUTPUT
            )
            self.message_log.append(envelope)

            return output

        def _get_parent_outputs(self, brain_id: str, flow_config: FlowConfig) -> list[BrainOutput]:
            """Get outputs from parent brains (dependencies)."""
            parent_ids = flow_config.nodes.get(brain_id, [])
            return [self.brain_outputs[pid] for pid in parent_ids if pid in self.brain_outputs]
    ```

    import pytest
    from mastermind_cli.orchestrator.stateless_coordinator import StatelessCoordinator
    from mastermind_cli.types.protocol import BrainEnvelope, BrainOutputType
    from mastermind_cli.types.parallel import FlowConfig
    from mastermind_cli.mcp.client import MCPClient

    @pytest.mark.asyncio
    async def test_brain_cascade_sequential():
        """Test Brain #1 → #3 → #7 cascade (parent outputs passed to children)."""

        # Mock MCP client
        class MockMCP:
            async def query_brain(self, brain_id: str, query: str, context: dict = None):
                # Simulate different outputs per brain
                outputs = {
                    "brain-01": {"content": "Product strategy defined", "metrics": {"quality": 0.9}},
                    "brain-03": {"content": "UX research complete", "users": 50},
                    "brain-07": {"content": "Evaluation approved", "verdict": "approve"}
                }
                return outputs.get(brain_id, {"content": "Default output"})

        coordinator = StatelessCoordinator(MockMCP())

        # Define flow: Brain #1 → #3 → #7 (sequential)
        flow_config = FlowConfig(
            flow_id="test-cascade",
            nodes={
                "brain-01": [],  # No dependencies
                "brain-03": ["brain-01"],  # Depends on #1
                "brain-07": ["brain-03"]   # Depends on #3
            }
        )

        # Execute flow
        response = await coordinator.execute_flow(
            brief="Test brief for cascade",
            brain_ids=["brain-01", "brain-03", "brain-07"],
            mcp_client=MockMCP()
        )

        # Verify execution order
        assert response.status == "success"
        assert len(coordinator.message_log) == 3

        # Verify Brain #3 received Brain #1's output
        msg_for_brain3 = coordinator.message_log[1]
        assert msg_for_brain3.message.from_brain == "brain-03"
        # Check context.parent_outputs contains brain-01 output

        # Verify Brain #7 received Brain #3's output
        msg_for_brain7 = coordinator.message_log[2]
        assert msg_for_brain7.message.from_brain == "brain-07"

    @pytest.mark.asyncio
    async def test_parallel_brains_no_dependencies():
        """Test independent brains execute in parallel."""

        class MockMCP:
            async def query_brain(self, brain_id: str, query: str, context: dict = None):
                await asyncio.sleep(0.1)  # Simulate 100ms delay
                return {"content": f"Output from {brain_id}"}

        coordinator = StatelessCoordinator(MockMCP())

        # Define flow: Brain #2, #5 independent (parallel)
        flow_config = FlowConfig(
            flow_id="test-parallel",
            nodes={
                "brain-02": [],  # No dependencies
                "brain-05": []   # No dependencies
            }
        )

        # Execute flow (should be faster than sequential)
        import time
        start = time.perf_counter()
        response = await coordinator.execute_flow(

            brain_ids=["brain-02", "brain-05"],

        elapsed = time.perf_counter() - start

        # Verify parallel execution (should be ~100ms, not 200ms)
