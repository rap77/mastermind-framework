         ```python
         async def refresh(request: RefreshRequest):
             # 1. Verify refresh token signature and expiry
             payload = jwt.decode(token, SECRET_KEY, algorithms=["HS256"])
             user_id = payload["sub"]

             # 2. Look up session by refresh_token_hash
             async with get_db() as db:
                 session = await db.execute(
                     "SELECT * FROM sessions WHERE user_id = ? AND refresh_token_hash = ?",
                     [user_id, hash_token(token)]
                 )
                 session = session.fetchone()

                 if not session:
                     raise HTTPException(401, "Invalid refresh token")

                 # 3. ROTATION: Delete old session, create new one
                 await db.execute(
                     "DELETE FROM sessions WHERE id = ?",
                     [session["id"]]
                 )

                 # 4. Generate NEW refresh token (invalidate old one)
                 new_refresh_token = create_refresh_token(user_id)
                 new_session_id = str(uuid.uuid4())

                 await db.execute(
                     "INSERT INTO sessions (id, user_id, refresh_token_hash, created_at, expires_at, rotation_count) VALUES (?, ?, ?, ?, ?, ?)",
                     [new_session_id, user_id, hash_token(new_refresh_token), datetime.utcnow(), datetime.utcnow() + timedelta(hours=24), session["rotation_count"] + 1]
                 )
                 await db.commit()

                 # 5. Return new access token AND new refresh token
                 new_access_token = create_access_token(user_id)
                 return {
                     "access_token": new_access_token,
                     "refresh_token": new_refresh_token,  # NEW token every time
                     "token_type": "Bearer"
                 }
         ```

         ```python
         class ThrottledBroadcaster:
             def __init__(self, interval_ms=300):
                 self.interval = interval_ms / 1000
                 self.accumulated = {}  # task_id -> [updates]
                 self.last_broadcast = {}  # task_id -> timestamp
                 self._lock = asyncio.Lock()

             async def add_update(self, task_id, update_data):
                 async with self._lock:
                     if task_id not in self.accumulated:
                         self.accumulated[task_id] = []
                     self.accumulated[task_id].append(update_data)

                     # Broadcast if interval elapsed
                     if time.time() - self.last_broadcast.get(task_id, 0) >= self.interval:
                         await self._flush(task_id)

             async def _flush(self, task_id):
                 updates = self.accumulated.get(task_id, [])
                 if not updates:
                     return

                 # Send batch to all connected clients
                 for ws in self.connections.get(task_id, set()):
                     try:
                         await ws.send_json({"type": "task_update_batch", "data": updates})
                     except:
                         pass  # Client disconnected, will clean up later

                 self.accumulated[task_id] = []
                 self.last_broadcast[task_id] = time.time()
         ```
