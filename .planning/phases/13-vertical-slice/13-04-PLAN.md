             }

             success: true,

           }

    1. Create docker/control-plane/Dockerfile:
       ```dockerfile
       FROM rust:1.75 as builder
       WORKDIR /app
       COPY Cargo.toml Cargo.lock ./
       COPY src ./src
       RUN cargo build --release

       FROM debian:bookworm-slim
       RUN apt-get update && apt-get install -y ca-certificates curl
       COPY --from=builder /app/target/release/mastermind-control-plane /usr/local/bin/
       EXPOSE 3001
       HEALTHCHECK --interval=10s --timeout=5s --retries=3 \
         CMD curl -f http://localhost:3001/health || exit 1
       CMD ["mastermind-control-plane"]
       ```

    2. Add control-plane service to docker-compose.yml:
       ```yaml
       control-plane:
         build:
           context: ./apps/control-plane
           dockerfile: ../../docker/control-plane/Dockerfile
         ports:
           - "3001:3001"
         environment:
           CONTROL_PLANE_URL: http://control-plane:3001
           AGENT_RUNTIME_URL: http://api:8001
           POSTGRES_URL: postgresql://postgres:devpassword@postgres:5432/mastermind_dev
         depends_on:
           postgres:
             condition: service_healthy
           api:
             condition: service_started
         healthcheck:
           test: ["CMD", "curl", "-f", "http://localhost:3001/health"]
           interval: 10s
           timeout: 5s
           retries: 3
       ```
