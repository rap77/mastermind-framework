```rust
use sqlx::PgPool;
use anyhow::Result;
use crate::event_sourcing::{BrainEvent, BrainEventType};

pub struct EventStore {
    pool: PgPool,
}

impl EventStore {
    pub fn new(pool: PgPool) -> Self {
        Self { pool }
    }

    pub async fn append_event(
        &self,
        brain_id: &str,
        event_type: BrainEventType,
        payload: serde_json::Value,
    ) -> Result<BrainEvent> {
        let event = BrainEvent {
            id: Uuid::new_v4(),
            brain_id: brain_id.to_string(),
            event_type,
            payload,
            created_at: Utc::now(),
        };

        sqlx::query!(
            "INSERT INTO activity_log (id, brain_id, event_type, payload, created_at)
             VALUES ($1, $2, $3, $4, $5)",
            event.id,
            event.brain_id,
            event.event_type as BrainEventType,
            event.payload,
            event.created_at,
        )
        .execute(&self.pool)
        .await?;

        Ok(event)
    }

    pub async fn read_events(
        &self,
        brain_id: Option<&str>,
        event_type: Option<BrainEventType>,
        start_time: Option<DateTime<Utc>>,
        end_time: Option<DateTime<Utc>>,
        limit: usize,
    ) -> Result<Vec<BrainEvent>> {
        let query = sqlx::query_as!(
            BrainEvent,
            "SELECT id, brain_id, event_type as \"event_type: BrainEventType\", payload, created_at
             FROM activity_log
             WHERE ($1::text IS NULL OR brain_id = $1)
               AND ($2::text IS NULL OR event_type = $2)
               AND ($3::timestamptz IS NULL OR created_at >= $3)
               AND ($4::timestamptz IS NULL OR created_at <= $4)
             ORDER BY created_at DESC
             LIMIT $5",
            brain_id,
            event_type.map(|t| t.to_string()),
            start_time,
            end_time,
            limit as i64,
        );

        let events = query.fetch_all(&self.pool).await?;
        Ok(events)
    }

    pub async fn replay_events(
        &self,
        session_id: Uuid,
    ) -> Result<Vec<BrainEvent>> {
        // Query events where payload->>'session_id' matches
        let events = sqlx::query_as!(
            BrainEvent,
            "SELECT id, brain_id, event_type as \"event_type: BrainEventType\", payload, created_at
             FROM activity_log
             WHERE payload->>'session_id' = $1
             ORDER BY created_at ASC",
            session_id.to_string(),
        )
        .fetch_all(&self.pool)
        .await?;

        Ok(events)
    }
}
```

```rust
    State(pool): State<PgPool>,
    Query(params): Query<ActivityLogQuery>,

        params.end_time,
```

2. Implement gRPC service:
```rust
use tonic::{Request, Response, Status};
use crate::proto::event_store::event_store_server::{EventStore as EventStoreService, EventStoreServer};
use crate::proto::event_store::*;

pub struct EventStoreGrpc {
    store: Arc<EventStore>,
}

impl EventStoreGrpc {
    pub fn new(store: Arc<EventStore>) -> Self {
        Self { store }
    }

    pub fn server(self) -> EventStoreServer<Self> {
        EventStoreServer::new(self)
    }
}

#[tonic::async_trait]
impl EventStoreService for EventStoreGrpc {
    async fn store_event(
        &self,
        request: Request<StoreEventRequest>,
    ) -> Result<Response<StoreEventResponse>, Status> {
        let req = request.into_inner();

        let event_type = match BrainEventType::try_from(req.event_type()) {
            Ok(t) => t,
            Err(_) => return Err(Status::invalid_argument("Invalid event type")),
        };

        let payload: serde_json::Value = serde_json::from_str(&req.payload)
            .map_err(|_| Status::invalid_argument("Invalid JSON payload"))?;

        let event = self.store.append_event(&req.brain_id, event_type, payload)
            .await
            .map_err(|_| Status::internal("Failed to store event"))?;

        Ok(Response::new(StoreEventResponse {
            success: true,
            event_id: event.id.to_string(),
        }))
    }
}
```
