use rusqlite::{Connection, Result as SqliteResult};
use anyhow::Result;

pub struct SqliteReader {
    conn: Connection,
}

impl SqliteReader {
    pub fn new(db_path: &str) -> Result<Self> {
        let conn = Connection::open(db_path)?;
        Ok(Self { conn })
    }

    pub fn get_row_counts(&self) -> Result<Vec<(String, usize)>> {
        let mut stmt = self.conn.prepare(
            "SELECT name, (SELECT COUNT(*) FROM sqlite_master WHERE name = m.name AND type = 'table') as count
             FROM sqlite_master m WHERE type = 'table' AND name NOT LIKE 'sqlite_%'"
        )?;
        let rows = stmt.query_map([], |row| {
            Ok((
                row.get::<_, String>(0)?,
                row.get::<_, usize>(1)?,
            ))
        })?;

        let mut counts = Vec::new();
        for row in rows {
            let (table, count) = row?;
            counts.push((table, count));
        }
        Ok(counts)
    }
}

pub async fn migrate_sqlite_to_postgresql(
    State(pool): State<PgPool>,
    State(sqlite_path): State<String>,
) -> Result<Json<MigrationStats>, StatusCode> {
    let reader = crate::sqlite_reader::SqliteReader::new(&sqlite_path)
        .map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)?;

    let tables = reader.get_row_counts()
        .map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)?;

    let mut migrated_tables = Vec::new();

    for (table_name, sqlite_count) in tables {
        // Read all rows from SQLite
        let sqlite_rows = reader.read_table(&table_name)
            .map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)?;

        // Migrate to PostgreSQL
        let migrated_count = crate::db::migration::migrate_table(&pool, &table_name, sqlite_rows)
            .await
            .map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)?;

        // Verify row count
        let pg_count: i64 = sqlx::query_scalar(&format!("SELECT COUNT(*) FROM {}", table_name))
            .fetch_one(&pool)
            .await
            .map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)?;

        let status = if pg_count as usize == sqlite_count {
            "migrated".to_string()
        } else {
            "partial".to_string()
        };

        migrated_tables.push(TableStats {
            table_name,
            sqlite_rows: sqlite_count,
            postgresql_rows: pg_count as usize,
            status,
        });
    }

    let total_rows = migrated_tables.iter().map(|t| t.postgresql_rows).sum();

    Ok(Json(MigrationStats {
        sqlite_path,
        tables: migrated_tables,
        total_rows,
    }))
}

use sqlx::PgPool;
use rusqlite::Connection;
use anyhow::Result;
use std::sync::Arc;

pub struct DualWriteRepository {
    pg_pool: Arc<PgPool>,
    sqlite_conn: Arc<Mutex<Connection>>,
}

impl DualWriteRepository {
    pub fn new(pg_pool: Arc<PgPool>, sqlite_path: &str) -> Result<Self> {
        let sqlite_conn = Connection::open(sqlite_path)?;
        Ok(Self {
            pg_pool,
            sqlite_conn: Arc::new(Mutex::new(sqlite_conn)),
        })
    }

    pub async fn write_task(&self, task: &Task) -> Result<()> {
        // Write to PostgreSQL (async)
        sqlx::query!(
            "INSERT INTO tasks (id, brain_id, status, progress, result, error, created_at, updated_at)
             VALUES ($1, $2, $3, $4, $5, $6, $7, $8)
             ON CONFLICT (id) DO UPDATE SET
                brain_id = EXCLUDED.brain_id,
                status = EXCLUDED.status,
                progress = EXCLUDED.progress,
                result = EXCLUDED.result,
                error = EXCLUDED.error,
                updated_at = EXCLUDED.updated_at",
            task.id,
            task.brain_id,
            task.status,
            task.progress,
            task.result,
            task.error,
            task.created_at,
            task.updated_at,
        )
        .execute(&*self.pg_pool)
        .await?;

        // Write to SQLite (sync)
        let conn = self.sqlite_conn.lock().unwrap();
        conn.execute(
            "INSERT OR REPLACE INTO tasks (id, brain_id, status, progress, result, error, created_at, updated_at)
             VALUES (?1, ?2, ?3, ?4, ?5, ?6, ?7, ?8)",
            [
                &task.id.to_string(),
                &task.brain_id,
                &task.status,
                &task.progress,
                &task.result,
                &task.error,
                &task.created_at,
                &task.updated_at,
            ],
        )?;

        Ok(())
    }

    pub async fn read_task(&self, id: Uuid) -> Result<Option<Task>> {
        // Read from PostgreSQL (after migration, primary source)
        let task = sqlx::query_as!(
            Task,
            "SELECT * FROM tasks WHERE id = $1",
            id
        )
        .fetch_optional(&*self.pg_pool)
        .await?;

        Ok(task)
    }
}
```

pub async fn verify_data_consistency(&self) -> Result<Vec<String>> {
    let tables = vec!["users", "sessions", "tasks", "executions", "api_keys", "experience_records"];
    let mut inconsistencies = Vec::new();

    for table in tables {
        let pg_count: i64 = sqlx::query_scalar(&format!("SELECT COUNT(*) FROM {}", table))
            .fetch_one(&*self.pg_pool)
            .await?;

        let conn = self.sqlite_conn.lock().unwrap();
        let sqlite_count: i64 = conn.query_row(&format!("SELECT COUNT(*) FROM {}", table), [], |row| row.get(0))?;

        if pg_count != sqlite_count {
            inconsistencies.push(format!(
                "{}: PostgreSQL={}, SQLite={}",
                table, pg_count, sqlite_count
            ));
        }
    }

    Ok(inconsistencies)
}

class DualWriteDatabaseConnection(DatabaseConnection):
    """Dual-write database connection manager (SQLite + PostgreSQL)."""

    def __init__(self, db_path: str = ":memory:", postgres_url: Optional[str] = None):
        super().__init__(db_path)
        self.postgres_url = postgres_url
        self._pg_conn: Optional[asyncpg.Connection] = None

    async def connect(self) -> None:
        """Connect to both SQLite and PostgreSQL."""
        await super().connect()  # SQLite connection

        if self.postgres_url:
            self._pg_conn = await asyncpg.connect(self.postgres_url)

    async def execute_write(self, sql: str, *args) -> None:
        """Write to both databases."""
        # Write to SQLite
        await self.conn.execute(sql, *args)

        # Write to PostgreSQL
        if self._pg_conn:
            await self._pg_conn.execute(sql, *args)

    async def execute_read(self, sql: str, *args, fetch_one: bool = False):
        """Read from PostgreSQL (primary source after migration)."""
        if self._pg_conn:
            if fetch_one:
                return await self._pg_conn.fetchrow(sql, *args)
            return await self._pg_conn.fetch(sql, *args)
        else:
            if fetch_one:
                return await self.conn.execute(sql, *args)
            return await self.conn.execute(sql, *args)

    async def verify_consistency(self) -> list[str]:
        """Verify row counts match between SQLite and PostgreSQL."""
        inconsistencies = []
        tables = ["users", "sessions", "tasks", "executions", "api_keys", "experience_records"]

        for table in tables:
            sqlite_count = await self.conn.fetchval(f"SELECT COUNT(*) FROM {table}")
            pg_count = await self._pg_conn.fetchval(f"SELECT COUNT(*) FROM {table}")

            if sqlite_count != pg_count:
                inconsistencies.append(f"{table}: SQLite={sqlite_count}, PostgreSQL={pg_count}")

        return inconsistencies
```
