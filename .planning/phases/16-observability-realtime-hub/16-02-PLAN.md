    use uuid::Uuid;
    use crate::tracing::metadata::TraceMetadata;

    pub async fn inject_trace_middleware(
        mut req: Request,
        next: Next,
    ) -> Response {
        let trace_id = Uuid::new_v4();
        let request_id = Uuid::new_v4();

        // Store in request extensions for handlers to access
        req.extensions_mut().insert(TraceMetadata {
            trace_id,
            request_id,
            user_id: None, // Extract from JWT if present
        });

        next.run(req).await
    }
    ```

    import structlog
    from grpclib.metadata import Metadata

    def extract_trace_context(metadata: Metadata) -> dict:
        """Extract trace_id from gRPC metadata and bind to logger context."""
        trace_id = metadata.get("trace-id", "unknown")

        logger = structlog.get_logger()
        logger = logger.bind(trace_id=trace_id)

        return {"trace_id": trace_id}
    ```
