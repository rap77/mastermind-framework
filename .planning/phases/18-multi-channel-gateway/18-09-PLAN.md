       ```rust
       use tonic::transport::Channel;
       use mastermind::worker::worker_client::WorkerClient;
       use mastermind::worker::ProcessWebhookRequest;

       pub struct AiWorkerClient {
           client: WorkerClient<Channel>,
       }

       impl AiWorkerClient {
           pub async fn new(addr: &str) -> Result<Self, Box<dyn std::error::Error>> {
               let client = WorkerClient::connect(addr.to_string()).await?;
               Ok(Self { client })
           }

           pub async fn process_webhook(
               &self,
               trace_id: String,
               channel: String,
               payload: String,
           ) -> Result<ProcessWebhookResponse, tonic::Status> {
               let request = ProcessWebhookRequest {
                   trace_id,
                   channel,
                   payload,
                   sender_id: "".to_string(),
                   message_type: "text".to_string(),
               };

               let mut client = self.client.clone();
               let response = client.process_webhook(request).await?;
               Ok(response.into_inner())
           }
       }
       ```

       ```rust
       use crate::grpc::AiWorkerClient;

       async fn send_to_ai_worker(&self, event: WebhookEvent) -> Result<String, Error> {
           let client = self.ai_worker_client.as_ref().ok_or("AI worker client not initialized")?;

           let response = client.process_webhook(
               event.trace_id.clone(),
               event.channel.clone(),
               event.payload.clone(),
           ).await.map_err(|e| Error::GrpcError(e.to_string()))?;

           if !response.success {
               return Err(Error::AiProcessingFailed(response.error_message));
           }

           Ok(response.ai_response)
       }
       ```

       ```python
       from mastermind.worker.worker_pb2 import ProcessWebhookResponse
       from mastermind.worker.worker_grpc import WorkerBase
       from routers.whatsapp import send_whatsapp_message
       from routers.instagram import send_instagram_message
       from routers.email import send_email

       class WorkerService(WorkerBase):
           async def ProcessWebhook(self, stream):
               req = await stream.recv()
               try:
                   # Route to appropriate channel sender
                   if req.channel == "whatsapp":
                       await send_whatsapp_message(req.payload)
                   elif req.channel == "instagram":
                       await send_instagram_message(req.payload)
                   elif req.channel == "email":
                       await send_email(req.payload)

                   return ProcessWebhookResponse(
                       success=True,
                       ai_response="Processed",
                       suggested_channel=req.channel,
                       processing_duration_ms=100
                   )
               except Exception as e:
                   return ProcessWebhookResponse(
                       success=False,
                       error_message=str(e)
       ```
